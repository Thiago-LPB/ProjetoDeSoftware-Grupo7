itrProducts = bancoDados.getProducts().iterator();
                    JPanel panel = new JPanel();
                    panel.setLayout(new BoxLayout(panel, BoxLayout.Y_AXIS));
                    JDialog f = new JDialog(frame, "Feed", true);
                    f.setLayout(new BorderLayout());

                    while(itrProducts.hasNext()){
                        JPanel itemPanel = new JPanel(new BorderLayout());
                        JButton buy = new JButton("comprar");
                        String str = itrProducts.toString();
                        itemPanel.add(new JLabel(str), BorderLayout.CENTER);
                        itemPanel.add(buy, BorderLayout.EAST);
                        panel.add(itemPanel);

                        buy.addActionListener(new ActionListener() {
                            @Override
                            public void actionPerformed(ActionEvent e) {
                                f.dispose();
                            }
                        });
                    }
                    JScrollPane scrollPane = new JScrollPane(panel);
                    scrollPane.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_ALWAYS);

                    JButton ok = new JButton("Fechar");

                    ok.addActionListener(h -> {
                        f.dispose();
                    });

                    f.add(scrollPane, BorderLayout.CENTER);
                    f.add(ok, BorderLayout.SOUTH);
                    f.pack();
                    f.setVisible(true);
                    f.setResizable(false);
